2** Snapshot system core data structures for the extensible SQL parser.
3**
4** A ParserSnapshot captures the complete state of a parser's tables at a
5** point in time. Snapshots use atomic reference counting so they can be
6** safely shared across threads: readers acquire a reference before use and
7** release it when done. When the last reference is released the snapshot
8** and all of its owned memory are freed.

103/*
104** A ParserSnapshot holds a frozen copy of every table the generated parser
105** needs at runtime. Fields fall into three groups:
106**
107** 1. Bookkeeping - version, refcount, timestamps
108** 2. Grammar data - symbols, rules, states (deep copies)
109** 3. Action tables - the compact arrays that drive the parse engine
110** 4. Module data - optional module identity and content hash
111*/
112typedef struct ParserSnapshot {
113 /* Monotonically increasing version number. Each new snapshot produced
114 ** from a grammar modification gets a higher version than the last. */
115 uint64_t version;
116
117 /* Atomic reference count. Starts at 1 on creation. Every call to
118 ** snapshot_acquire() adds 1; every call to snapshot_release() subtracts
119 ** 1. When the count drops to 0 the snapshot is destroyed. */
120 atomic_uint_fast32_t refcount;
121
122 /* --- Grammar data (deep-copied, owned by this snapshot) ----------- */
123
124 struct symbol **symbols; /* Array of pointers to symbol structs */
125 uint32_t nsymbol; /* Total number of symbols */
126 uint32_t nterminal; /* Number of terminal symbols */
127
128 struct rule *rules; /* Linked list of production rules */
129 uint32_t nrule; /* Total number of rules */
130
131 struct state **states; /* Array of pointers to state structs */
132 uint32_t nstate; /* Total number of parser states */
133
134 /* --- Compact action tables (heap-allocated, owned) ---------------- */
135
136 uint16_t *yy_action; /* Combined shift+reduce action array */
137 uint16_t *yy_lookahead; /* Lookahead values parallel to yy_action*/
138 int16_t *yy_shift_ofst; /* Per-state offset into yy_action for shifts */
139 int16_t *yy_reduce_ofst; /* Per-state offset into yy_action for reduces */
140 uint16_t *yy_default; /* Default action for each state */
141 uint32_t action_count; /* Number of entries in yy_action */
142 uint32_t lookahead_count; /* Number of entries in yy_lookahead */
143
144 /* --- Module identity (optional, NULL when not part of a module) --- */
145
146 uint8_t merkle_root[32]; /* Content hash of grammar data */
147 ParserModule *module; /* Owning module metadata (or NULL) */
148
149 /* Nanosecond-precision wall-clock time when this snapshot was created */
150 uint64_t create_time_ns;
151
152 /* Reserved for a future JIT compilation context that can cache
153 ** machine code generated from the action tables. */
154 void *jit_ctx;
155} ParserSnapshot;

157/*
158** Acquire a reference to an existing snapshot. The caller must eventually
159** call snapshot_release() to avoid leaking the snapshot.
160**
161** Returns the same pointer that was passed in, for convenience:
162** ParserSnapshot *my_ref = snapshot_acquire(shared_snap);
163**
164** Passing NULL is safe and returns NULL.
165*/
166ParserSnapshot *snapshot_acquire(ParserSnapshot *snap);

168/*

175void snapshot_release(ParserSnapshot *snap);
