1/*
2** JIT compilation policy for the extensible SQL parser.
3**
4** Decides when to JIT compile a snapshot based on runtime metrics.
5** The policy tracks per-snapshot usage statistics and triggers
6** background JIT compilation when the expected benefit (faster action
7** table lookups) exceeds the compilation cost.
8**
9** Thread safety: JITMetrics use atomic counters so multiple parser
10** threads can update them concurrently. The background compilation
11** thread is managed internally and publishes the JIT context to the
12** snapshot via an atomic pointer store.
13*/

32/*
33** Per-snapshot runtime metrics used by the JIT policy to decide
34** whether compilation is worthwhile. All counters are atomic so
35** they can be updated from any parser thread without locking.
36*/
37typedef struct JITMetrics {
38 atomic_uint_fast64_t parse_count; /* Number of parse sessions */
39 atomic_uint_fast64_t total_parse_time_ns; /* Cumulative parse wall-clock */
40 atomic_uint_fast64_t action_lookup_count; /* Total action table lookups */
41 atomic_uint_fast64_t total_tokens_parsed; /* Cumulative tokens across all parses */
42 atomic_int is_jitted; /* 1 if JIT code is attached */
43 atomic_int jit_in_progress; /* 1 if background compile active */
44} JITMetrics;

50/*
51** Tunable thresholds for the JIT compilation policy.
52** These can be adjusted per-application to match workload patterns.
53*/
54typedef struct JITPolicyConfig {
55 /* Minimum number of parse sessions before considering JIT.
56 ** Prevents wasting compilation effort on rarely-used grammars. */
57 uint64_t min_parse_count;
58
59 /* Minimum cumulative parse time (nanoseconds) before JIT.
60 ** Ensures the grammar is spending enough time in parsing to
61 ** justify the compilation overhead. */
62 uint64_t min_total_parse_time_ns;
63
64 /* Minimum average action lookups per parse session.
65 ** Grammars with few lookups per parse won't benefit from JIT. */
66 uint64_t min_avg_lookups_per_parse;
67
68 /* If true, JIT compilation happens on a background thread.
69 ** If false, compilation is synchronous (blocks the caller). */
70 bool background_compile;
71
72 /* Minimum number of parser states before considering JIT.
73 ** Small grammars don't benefit enough to justify compilation. */
74 uint32_t min_grammar_states;
75
76 /* Minimum average tokens per parse session before JIT.
77 ** Short inputs don't benefit from JIT compilation. */
78 uint32_t min_avg_tokens_per_parse;
79
80 /* Master switch for JIT compilation.
81 ** Set to false to disable JIT entirely regardless of metrics. */
82 bool enabled;
83
84 /* Enable JIT compilation of the keyword tokenizer.
85 ** When true, the JIT policy will compile a trie-based keyword
86 ** classifier from the TokenTable when the parser JIT triggers.
87 ** When false, keyword lookups always use the hash-based path. */
88 bool tokenizer_jit_enabled;
89} JITPolicyConfig;

139/*
140** Possibly trigger JIT compilation for a snapshot.
141**
142** Checks the metrics against the policy; if compilation is warranted:
143** - If config->background_compile is true, spawns a detached thread
144** that compiles and atomically publishes the JIT context.
145** - If config->background_compile is false, compiles synchronously.
146**
147** Returns:
148** 0 -- compilation triggered (or already done)
149** 1 -- metrics do not yet warrant compilation
150** -1 -- error (JIT unavailable, snapshot NULL, etc.)
151**
152** Thread-safe: uses the jit_in_progress flag to prevent concurrent
153** compilation attempts on the same snapshot.
154*/

156 JITMetrics *m,
157 const JITPolicyConfig *config);
